# Calculate z-scores for daily call counts
daily_calls <- daily_calls |>
  mutate(
    mean_calls = mean(call_count),
    sd_calls = sd(call_count),
    z_score = (call_count - mean_calls) / sd_calls
  )
---> the formula for z-score is "(the data point - mean)/SD"
